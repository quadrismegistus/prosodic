140    def to_html(self, parse: Optional[Any] = None, as_str: bool = False, css: str = HTML_CSS, tooltip: bool = False, **kwargs) -> Any:
141        """
142        Generate an HTML representation of the Line.
143
144        Args:
145            parse (Optional[Any]): The parse to use for rendering. If None, uses the minimum parse.
146            as_str (bool): If True, return the result as a string. Default is False.
147            css (str): CSS styles to include in the HTML. Default is HTML_CSS.
148            tooltip (bool): Whether to include tooltips. Default is False.
149            **kwargs: Additional keyword arguments.
150
151        Returns:
152            Any: HTML representation of the Line.
153        """
154        if parse is None:
155            parse = min(self._parses)
156
157        output = []
158
159        for wordtoken in self.wordtokens:
160            prefstr = get_initial_whitespace(wordtoken.txt)
161            if prefstr:
162                odx = {"txt": prefstr}
163                output.append(odx)
164
165            wordtoken_slots = parse.wordtoken2slots[wordtoken]
166            if wordtoken_slots:
167                for slot in wordtoken_slots:
168                    pos = slot.parent
169                    spclass = f"mtr_{'s' if slot.is_prom else 'w'}"
170                    stclass = f"str_{'s' if slot.unit.is_stressed else 'w'}"
171                    vclass = f"viol_{'y' if pos.violset else 'n'}"
172                    odx = {
173                        "txt": slot.unit.txt,
174                        "meter": spclass,
175                        "stress": stclass,
176                        "viol": vclass,
177                        "viols": list(slot.violset),
178                    }
179                    output.append(odx)
180            else:
181                odx = {"txt": wordtoken.txt}
182                output.append(odx)
183
184        odf = pd.DataFrame(output)
185        odf = odf.fillna(method="ffill")
186
187        def htmlx(row, tooltip=tooltip):
188            if not row.txt.strip() or not row.txt[0].isalpha():
189                return row.txt
190
191            if tooltip and row.viols:
192                viol_strs = [f"<li>{viol}</li>" for viol in sorted(row.viols)]
193                viol_str = f'<ol>{"".join(viol_strs)}</ol>'
194                viol_title = f"Violated {len(row.viols)} constraints: {viol_str}"
195                rowtxt = f'{row.txt}<span class="tooltip">{viol_title}</span>'
196                tooltip = " tooltip"
197            else:
198                tooltip = ""
199                rowtxt = row.txt
200
201            return f'<span class="{row.meter} {row.stress} {row.viol}{tooltip}">{rowtxt}</span>'
202
203        spans = odf.apply(htmlx, axis=1)
204        out = "".join(spans)
205        out = f'<style>{css}</style><div class="parse">{out}</div>'
206        return to_html(out, as_str=as_str)

261class LineList(EntityList):
262    def get_rhyming_lines(self, max_dist=RHYME_MAX_DIST):
263        line2rhyme = defaultdict(list)
264        for line in self.data:
265            prev_lines = self.data[: line.i]
266            if not prev_lines:
267                continue
268            for line2 in prev_lines:
269                dist = line.rime_distance(line2)
270                if max_dist is None or dist <= max_dist:
271                    line2rhyme[line].append((dist, line2))
272        return {i: min(v) for i, v in line2rhyme.items()}
