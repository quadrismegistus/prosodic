219    def bound(self, progress: bool = False) -> 'ParseList':
220        """
221        Bound the parses in this ParseList.
222
223        Args:
224            progress: Whether to show progress during bounding.
225
226        Returns:
227            A ParseList containing unbounded parses after bounding.
228        """
229        parses = [p for p in self.data if not p.is_bounded]
230        iterr = tqdm(parses, desc="Bounding parses", disable=not progress, position=0)
231        for parse_i, parse in enumerate(iterr):
232            parse.constraint_viols  # init
233            if parse.is_bounded:
234                continue
235            for comp_parse in parses[parse_i + 1 :]:
236                if comp_parse.is_bounded:
237                    continue
238                if not parse.can_compare(comp_parse):
239                    continue
240                relation = parse.bounding_relation(comp_parse)
241                if relation == Bounding.bounded:
242                    parse.is_bounded = True
243                    parse.bounded_by.append(
244                        (comp_parse.meter_str, comp_parse.stress_str)
245                    )
246                elif relation == Bounding.bounds:
247                    comp_parse.is_bounded = True
248                    comp_parse.bounded_by.append((parse.meter_str, parse.stress_str))
249        self._bound_init = True
250        return self.unbounded

324        Returns:

337    def _get_aggby(self, df: pd.DataFrame) -> Dict[str, str]:
338        """
339        Get the aggregation functions for statistics.
340
341        Args:
342            df: The DataFrame to aggregate.
343
344        Returns:
345            A dictionary mapping column names to aggregation functions.
346        """
347        df_q = df.select_dtypes("number")
348
349        def getagg(col):
350            if col.endswith("_norm"):
351                return "mean"
352            if self.type in {"text", "stanza"}:
353                return "median"
354            return "median"
355
356        aggby = {col: getagg(col) for col in df_q}

360    def stats(self, norm: Optional[bool] = None, incl_bounded: Optional[bool] = None, by: Optional[str] = None, **kwargs: Any) -> pd.DataFrame:
361        """
362        Get statistics for this ParseList as a DataFrame.
363
364        Args:
365            norm: Whether to normalize the statistics.
366            incl_bounded: Whether to include bounded parses.
367            by: How to group the statistics.
368            **kwargs: Additional keyword arguments.
369
370        Returns:
371            A DataFrame containing statistics.
372        """
373        if incl_bounded is None:
374            incl_bounded = self.show_bounded
375        if by == "syll":
376            odf = self.df_syll
377            if not incl_bounded:
378                odf = odf[odf.parse_is_bounded == 0]
379            return odf
380        odf = pd.DataFrame(
381            parse.stats_d(norm=norm)
382            for parse in self
383            if incl_bounded or not parse.is_bounded
384        )
385        odf.columns = [
386            (
387                f"parse_{c}"
388                if c[0] != "*" and c.split("_")[0] not in {"stanza", "line"}
389                else c
390            )
391            for c in odf
392        ]
393        groupby = self._get_groupby(by=by)
394        if groupby:
395            odf = odf.set_index(groupby)
396            aggby = self._get_aggby(odf)
397            odf = odf.groupby(groupby).agg(aggby)
398            odf = odf.drop("parse_rank", axis=1)
399            if not "line_num" in set(groupby):
400                odf = odf.drop("line_num", axis=1)
401            return odf.sort_index()
402        else:
403            odf["parse_rank"] = (
404                odf.groupby(["stanza_num", "line_num"])
405                .parse_rank.rank(method="min")
406                .apply(force_int)
407            )
408            return setindex(odf, DF_INDEX).sort_index()

221        """
222        Bound the parses in this ParseList.
223
224        Args:

228            A ParseList containing unbounded parses after bounding.
229        """
230        parses = [p for p in self.data if not p.is_bounded]
231        iterr = tqdm(parses, desc="Bounding parses", disable=not progress, position=0)

233            parse.constraint_viols  # init
234            if parse.is_bounded:
