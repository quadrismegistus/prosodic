248    @cached_property
249    def rime(self) -> Optional['PhonemeList']:
250        """
251        Get the rime of the word form.
252
253        Returns:
254            PhonemeList: The rime of the word form, or None if no rime can be determined.
255        """
256        from .phonemes import PhonemeList
257
258        sylls = []
259        for syll in reversed(self.children):
260            sylls.insert(0, syll)
261            if syll.stress == "P":
262                break
263        if not sylls:
264            return
265        o = sylls[0].rime.data + [phon for syll in sylls[1:] for phon in syll.children]
266        return PhonemeList(o)

268    @cache
269    def rime_distance(self, wordform: 'WordForm') -> float:
270        """
271        Calculate the rime distance between this word form and another.
272
273        Args:
274            wordform (WordForm): The word form to compare with.
275
276        Returns:
277            float: The rime distance between the two word forms.
278        """
279        from scipy.spatial.distance import euclidean
280
281        if self.txt == wordform.txt:
282            return np.nan
283        # return self.syllables[-1].rime_distance(wordform.syllables[-1])
284
285        df1 = self.rime.df
286        df2 = wordform.rime.df
287
288        if list(df1.reset_index().phon_txt) == list(df2.reset_index().phon_txt):
289            return 0
290
291        s1 = df1.mean(numeric_only=True)
292        s2 = df2.mean(numeric_only=True)
293        keys = [
294            k
295            for k in set(s1.index) & set(s2.index)
296            if k.startswith("phon_")
297            and not k.endswith("_num")
298            and not k.startswith("phon_is_")
299        ]
300        s1x = s1.loc[keys]
301        s2x = s2.loc[keys]
302        dist = euclidean(s1x, s2x)
303        return dist
