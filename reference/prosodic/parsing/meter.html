217    def parse_line(self, line: Line, force: bool = False, **kwargs: Any) -> ParseList:
218        """
219        Parse a line.
220
221        Args:
222            line (Line): The line to parse.
223            force (bool, optional): Force parsing even if cached. Defaults to False.
224            **kwargs: Additional keyword arguments.
225
226        Returns:
227            ParseList: List of parses for the line.
228        """
229        assert line.is_parseable
230        if line.num_sylls < 2:
231            return ParseList(type="line", line=line)

234            parses = self.parses_from_cache(line)
235            line._parses = parses
236            if parses:
237                return parses

241        else:
242            parses = self.parse_line_fast(line)

244        if self.use_cache_lines and caching_is_enabled():
245            self.cache(val_obj=parses, key=self.get_key(line))

247        return parses

292    def parse_line_fast(self, line: Line, force: bool = False) -> ParseList:
293        """
294        Parse a line using the fast parsing method.
295
296        Args:
297            line (Line): The line to parse.
298            force (bool, optional): Force parsing even if cached. Defaults to False.
299
300        Returns:
301            ParseList: List of parses for the line.
302        """
303        from .parses import Parse
304        from .parselists import ParseList
305
306        assert line.is_parseable
307        parses = ParseList(
308            [
309                Parse(
310                    wfl,
311                    pos,
312                    meter=self,
313                    parent=line,
314                )
315                for wfl in self.get_wordform_matrix(line)
316                for pos in self.get_pos_types(nsylls=wfl.num_sylls)
317            ],
318            line=line,
319        )
320        for n in range(1000):
321            # logger.debug(f'Now at {i}A, there are {len(parses)} parses')
322            parses = ParseList(
323                [
324                    newparse
325                    for parse in parses
326                    for newparse in parse.branch()
327                    if not parse.is_bounded
328                    and newparse is not None
329                    and parse is not None
330                ],
331                type="line",
332                line=line,
333            )
334            parses.bound(progress=False)
335            if all(p.is_complete for p in parses):
336                break
337        else:
338            logger.error(f"did not complete parsing: {line}")
339        parses.bound(progress=False)
340        parses.rank()
341        line._parses = parses
342        return line._parses

346    def parse_line_exhaustive(self, line: Line, progress: Optional[bool] = None) -> ParseList:
347        """
348        Parse a line using the exhaustive parsing method.
349
350        Args:
351            line (Line): The line to parse.
352            progress (bool, optional): Whether to show progress. Defaults to None.
353
354        Returns:
355            ParseList: List of parses for the line.
356        """
357        from .parses import Parse
358        from .parselists import ParseList
359
360        assert line.is_parseable
361
362        def iter_parses():
363            wfm = self.get_wordform_matrix(line)
364            all_parses = []
365            combos = [
366                (wfl, scansion)
367                for wfl in wfm
368                for scansion in get_possible_scansions(
369                    wfl.num_sylls, max_s=self.max_s, max_w=self.max_w
370                )
371            ]
372            wfl = wfm[0]
373            logger.trace(
374                f"Generated {len(combos)} from a wordfrom matrix of size {len(wfm), wfl, wfl.num_sylls, self.max_s, self.max_s, len(get_possible_scansions(wfl.num_sylls))}"
375            )
376            iterr = tqdm(combos, disable=not progress, position=0)
377            for wfl, scansion in iterr:
378                parse = Parse(wfl, scansion, meter=self, parent=line)
379                all_parses.append(parse)
380            logger.trace(f"Returning {len(all_parses)} parses")
381            return all_parses
382
383        parses = ParseList(iter_parses(), type="line", line=line)
384        parses.bound(progress=False)
385        parses.rank()
386        line._parses = parses
387        return line._parses

522        Helper function for parsing a text in parallel processing.
