212    def stresses(self, leaves=True, arto=False):
213        """"""
214
215        for preterminal in self.preterminals(leaves=True):
216            if leaves:

218                    if preterminal._stress is None:

220                    elif preterminal._lstress == -1:

250    def set_lstress(self):
251        """"""
252
253        if self._preterm:
254            if self[0].lower() in super(MetricalTree, self)._contractables:
255                self._lstress = np.nan
256            elif self._cat in super(MetricalTree, self)._punctTags:
257                self._lstress = np.nan

259            elif self[0].lower() in MetricalTree._unstressedWords:
260                self._lstress = -1
261            elif self[0].lower() in MetricalTree._ambiguousWords:
262                self._lstress = -.5
263            elif self[0].lower() in MetricalTree._stressedWords:
264                self._lstress = 0
265
266            elif self._cat in MetricalTree._unstressedTags:
267                self._lstress = -1
268            elif self._cat in MetricalTree._ambiguousTags:
269                self._lstress = -.5

271            elif self._dep in MetricalTree._unstressedDeps:
272                self._lstress = -1
273            elif self._dep in MetricalTree._ambiguousDeps:
274                self._lstress = -.5

276            else:
277                self._lstress = 0

279            if self[0].lower() == 'that' and (self._cat == 'DT' or self._dep == 'det'):
280                self._lstress = -.5
281        else:
282            for child in self:
283                child.set_lstress()

287    # Set the phrasal stress of the tree
288    def set_pstress(self):
289        """"""
290
291        # Basis
292        if self._preterm:
293            try: assert self._lstress != -.5
294            except: raise ValueError('The tree must be disambiguated before assigning phrasal stress')
295            self._pstress = self._lstress
296        else:
297            # Recurse
298            for child in self:
299                child.set_pstress()
300            assigned = False
301            # Noun compounds (look for sequences of N*)
302            if self._cat == 'NP':
303                skipIdx = None
304                i = len(self)
305                for child in self[::-1]:
306                    i -= 1
307                    if child._cat.startswith('NN'):
308                        if not assigned and skipIdx is None:
309                            skipIdx = i
310                            child._pstress = -1
311                            child.set_label()
312                        elif not assigned:
313                            child._pstress = 0
314                            child.set_label()
315                            assigned = True
316                        else:
317                            child._pstress = -1
318                            child.set_label()
319                    elif assigned:
320                        child._pstress = -1
321                        child.set_label()
322                    else:
323                        if not assigned and skipIdx is not None:
324                            self[skipIdx]._pstress = 0
325                            self[skipIdx].set_label()
326                            assigned = True
327                            child._pstress = -1
328                            child.set_label()
329                        else:
330                            break
331                if not assigned and skipIdx is not None:
332                    self[skipIdx]._pstress = 0
333                    self[skipIdx].set_label()
334                    assigned = True
335            # Everything else
336            if not assigned:
337                for child in self[::-1]:
338                    if not assigned and child._pstress == 0:
339                        assigned = True
340                    elif not np.isnan(child._pstress):
341                        child._pstress = -1
342                        child.set_label()
343            if not assigned:
344                self._pstress = -1
345            else:
346                self._pstress = 0
347        self.set_label()

351    def set_stress(self, stress=0):
352        """"""
353
354        self._stress = self._lstress + self._pstress + stress
355        if not self._preterm:
356            for child in self:
357                child.set_stress(self._stress)

379    def convert(cls, tree):
380        """
381        Convert a tree between different subtypes of Tree.  ``cls`` determines
382        which class will be used to encode the new tree.
383
384        :type tree: Tree
385        :param tree: The tree that should be converted.
386        :return: The new Tree.
387        """
388
389        if isinstance(tree, Tree):
390            children = [cls.convert(child) for child in tree]
391            if isinstance(tree, MetricalTree):
392                return cls(tree._cat, children, tree._dep, tree._lstress)
393            elif isinstance(tree, DependencyTree):
394                return cls(tree._cat, children, tree._dep)
395            else:
396                return cls(tree._label, children)

398            return tree
