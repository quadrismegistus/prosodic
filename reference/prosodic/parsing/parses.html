237    def extend(self, mpos_str: str) -> Optional["Parse"]:
238        """
239        Extend the parse with a new metrical position.
240
241        Args:
242            mpos_str (str): String representation of the metrical position to add.
243
244        Returns:
245            Optional[Parse]: The extended parse, or None if extension is not possible.
246        """
247        slots = []
248        mval = mpos_str[0]
249        if self.positions and self.positions[-1].meter_val == mval:
250            # logger.warning(f'cannnot extend because last position is also {mval}')
251            return None
252
253        for i, x in enumerate(mpos_str):
254            slot_i = self.num_slots_positioned
255            try:
256                slot = self.wordforms.slots[slot_i]
257            except IndexError:
258                # logger.warning('cannot extend further, already taking up all syllable slots')
259                return None
260
261            slots.append(ParseSlot(slot, num=slot_i + 1))
262            self.num_slots_positioned += 1
263
264        mpos = ParsePosition(
265            meter_val=mval, children=slots, parent=self, num=len(self.positions) + 1
266        )
267        self.positions.append(mpos)
268        self.constraint_viols  # init and bound
269        return self

304    def branch(self) -> List["Parse"]:
305        """
306        Create branching parses from this parse.
307
308        Returns:
309            List[Parse]: List of branching parses.
310        """
311        if self.is_bounded:
312            return []
313        if not self.positions:
314            logger.error("needs to start with some positions")
315            return
316        mval = self.positions[-1].meter_val
317        otypes = self.meter.get_pos_types(self.wordforms.num_sylls)
318        otypes = [x for x in otypes if x[0] != mval]
319        o = [copy(self).extend(posstr) for posstr in otypes]
320        o = [x for x in o if x is not None]
321        o = o if o else [self]
322        o = [p for p in o if not p.is_bounded]
323        return o

421        """
422        if (self.stanza_num, self.line_num) != (other.stanza_num, other.line_num):

425        if min_slots and self.num_slots_positioned < min_slots:

429            return True
430
431        if self.num_slots_positioned != other.num_slots_positioned:
432            return False
433
434        return True
