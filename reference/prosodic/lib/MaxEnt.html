  4# class WeightingAnalysis ================
  5# ========================================
  6# This class serves to provide an analysis
  7# of the importants of various constraints
  8# in determination of the "correct" parse
  9#
 10# Given a set of hand-annotated lines,
 11# this class analyzes the set of possible
 12# parses for each line to determine the
 13# possible interpretations. Then, it
 14# uses a Maximum Entropy optimization
 15# algorithm to determine the best weighting
 16# for each contstraint so as to maximize the
 17# probability of the hand-selected parse(s)
 18# while minimizing the probability of the
 19# non-selected parses.
 20
 21# Based on algorith and mathematics as
 22# described here: http://homepages.inf.ed.ac.uk/sgwater/papers/OTvar03.pdf
 23# and loosely informed by MEGrammarTool from Bruce Hayes

349    def calculate_gradient(self):
350        gradient = -self.calculate_overfit_penalty_gradient()
351        for key in self.data:
352            outs, freqs = self.data[key]
353            quotient = self.calculate_gradient_quotient_term(outs)
354            difference = outs - quotient
355
356            unsummed_gradient = freqs[:, None] * difference
357            gradient += np.sum(unsummed_gradient, axis=0)
358        return gradient
359
360    def calculate_gradient_quotient_term(self, outs):
361        scores = np.matmul(outs, self.weights)
362        exp_scores = np.exp(scores)
363
364        denominator = np.sum(exp_scores)
365        unsummed_numerator = outs * exp_scores[:, None]
366        numerator = np.sum(unsummed_numerator, axis = 0)
367
368        return numerator / denominator
369
370    def calculate_overfit_penalty_gradient(self):
371        return (self.weights - self.mu) / self.sigma
372
373    def calculate_objective(self):
374        weight_penalty = self.calculate_overfit_penalty()
375
376        total = 0.0
377        for key in self.data:
378            out, freqs = self.data[key]
379
380            probs = self.calculate_probabilities(out)
381            log_probs = np.log(probs)
382            total += np.sum(freqs * log_probs)
383
384        return total - weight_penalty
385
386    def calculate_probabilities(self, out):
387        scores = np.matmul(out, self.weights)
388        exp_scores = np.exp(scores)
389        denominator = np.sum(exp_scores)
390
391        return exp_scores / denominator
392
393    def calculate_overfit_penalty(self):
394        numerator = (self.weights - self.mu)**2
395        denominator = 2 * self.sigma
396
397        return np.sum(numerator / denominator)
