 41class LineList(EntityList):
 42    def get_rhyming_lines(self, max_dist=RHYME_MAX_DIST):
 43        line2rhyme = defaultdict(list)
 44        for line in self.data:
 45            prev_lines = self.data[: line.i]
 46            if not prev_lines:
 47                continue
 48            for line2 in prev_lines:
 49                dist = line.rime_distance(line2)
 50                if max_dist is None or dist <= max_dist:
 51                    line2rhyme[line].append((dist, line2))
 52        return {i: min(v) for i, v in line2rhyme.items()}

161    def bound(self, progress=False):
162        parses = [p for p in self.data if not p.is_bounded]
163        iterr = tqdm(parses, desc="Bounding parses", disable=not progress, position=0)
164        for parse_i, parse in enumerate(iterr):
165            parse.constraint_viols  # init
166            if parse.is_bounded:
167                continue
168            for comp_parse in parses[parse_i + 1 :]:
169                if comp_parse.is_bounded:
170                    continue
171                if not parse.can_compare(comp_parse):
172                    continue
173                relation = parse.bounding_relation(comp_parse)
174                if relation == Bounding.bounded:
175                    parse.is_bounded = True
176                    parse.bounded_by.append(
177                        (comp_parse.meter_str, comp_parse.stress_str)
178                    )
179                elif relation == Bounding.bounds:
180                    comp_parse.is_bounded = True
181                    comp_parse.bounded_by.append((parse.meter_str, parse.stress_str))
182        self._bound_init = True
183        return self.unbounded
